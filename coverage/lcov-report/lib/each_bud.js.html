"use strict";
 
const expandglob = require('expandglob'),
    path = require('path'),

/** @lends eachBud */
function eachBud(pattern, handler, callback) {
    async.concatSeries([].concat(pattern), (pattern, callback) => {
        switch (typeof(pattern)) {
            case 'object':
                var bud = pattern; //Actually a bud.
                bud.cwd = bud.cwd || _cwd();
                handler(bud, callback);
                break;
            case 'function':
                async.waterfall([
                    (callback) => {
                        pattern(callback);
                    },
                    (buds, callback) => {
                        eachBud(buds, handler, callback);
                    }
                ], callback);
                break;
            default:
                _eachFiles([].concat(pattern), handler, callback);
                break;
        }
    }, callback);
}
